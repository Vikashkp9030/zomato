type ticketRequest struct {
	personId   int
	numTickets int
	cost       int
}

func ticketSeller(requests <-chan ticketRequest, results chan<- int) {
	for req := range requests {
		fmt.Printf("Processing %d tickets(s) of personId %d with total cast %d\n", req.numTickets, req.personId, req.cost)
		time.Sleep(time.Second)
		results <- req.personId
	}
}

func main() {
	numberOfSellers := 3
	price := 5
	requests := make(chan ticketRequest, 10)
	tickerResult := make(chan int)

	for range 3 {
		go ticketSeller(requests, tickerResult)
	}

	for i := range numberOfSellers {
		requests <- ticketRequest{personId: i, numTickets: 2 * i, cost: price * (i + 1)}
	}
	close(requests)
	for range requests {
		fmt.Printf("Ticket for personId %d processed successfully\n", <-tickerResult)
	}
}
