package main

import "fmt"

func main() {
	fac := factorial(10)
	fmt.Println("factorial is :", fac)
	sum := sumDigits(12534)
	fmt.Println(sum)
}

func factorial(n int) int {
	if n <= 0 {
		return 1
	}
	return n * (factorial(n - 1))
}

func sumDigits(n int) int {
	if n < 10 {
		return n
	}
	return n%10 + sumDigits(n/10)
}
