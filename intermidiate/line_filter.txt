package main

import (
	"bufio"
	"fmt"
	"os"
	"strings"
)

func main() {
	file, err := os.Open("example.txt")
	if err != nil {
		fmt.Println("Error opening file")
		return
	}
	defer file.Close()
	keywords := "important"
	scanner := bufio.NewScanner(file)
	for scanner.Scan() {
		line := scanner.Text()
		if strings.Contains(line, keywords) {
			updateLine := strings.ReplaceAll(line, keywords, "necessary")
			fmt.Println("Line:", line)
			fmt.Println("Update Line:", updateLine)
		}
	}

	err = scanner.Err()
	if err != nil {
		fmt.Println("Error Scaning file:", err)
		return
	}
}
