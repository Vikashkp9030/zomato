package main

import (
	"fmt"
	"os"
)

func main() {
	file, err := os.Create("output.txt")

	if err != nil {
		fmt.Println("Error creating file.", file)
		return
	}
	defer file.Close()
	data := []byte("Hello world!\n\n\n\nhello ji")
	_, err = file.Write(data)
	if err != nil {
		fmt.Println("Error writing to file:", err)
		return
	}
	fmt.Println("Data has been written to file successfully")
	file, err = os.Create("writeString.txt")
	if err != nil {
		fmt.Println("Error Creating file:", err)
		return
	}
	defer file.Close()
	_, err = file.WriteString("Hello baby ji!")
	if err != nil {
		fmt.Println("Error writing to file:", err)
		return
	}
	fmt.Println("Write string successfully!")

}
