package main

import (
	"fmt"
)

func main() {
	a := [5]int{1, 3, 4, 5, 7}
	slice := a[1:4]
	fmt.Println(slice)
	slice = append(slice, 10, 20)
	fmt.Println(slice)
	sliceCopy := make([]int, len(slice))
	copy(sliceCopy, slice)
	fmt.Println(sliceCopy)
	for index, value := range slice {
		fmt.Println(index, value)
	}
	for _, value := range sliceCopy {
		fmt.Println(value)
	}
}
