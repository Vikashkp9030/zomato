package main

import "fmt"

func main() {
	var a, b int = 10, 3
	var result int = a + b
	fmt.Println("Addition:", result)

	result = a - b
	fmt.Println("Substractions:", result)

	result = a * b
	fmt.Println("Multiplication:", result)

	result = a / b
	fmt.Println("Division:", result)

	result = a % b
	fmt.Println("Modulus:", result)
}
